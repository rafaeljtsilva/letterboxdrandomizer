async function carregarWatchlist() { const username = document.getElementById('username').value.trim().toLowerCase(); if (!username) return alert("Escreve o teu username!"); document.getElementById('loading').style.display = 'block'; document.getElementById('lista-filmes').innerHTML = ''; // URL da página normal da tua watchlist (sem RSS!) const urlWatchlist = `https://letterboxd.com/${username}/watchlist/`; // Usamos o proxy AllOrigins para trazer o HTML bruto da página para o nosso código const urlProxy = `https://api.allorigins.win/get?url=${encodeURIComponent(urlWatchlist)}`; try { const resposta = await fetch(urlProxy); if (!resposta.ok) throw new Error(); const dados = await resposta.json(); const htmlBruto = dados.contents; // Criamos um elemento virtual para conseguir ler o HTML como se fosse uma página const parser = new DOMParser(); const doc = parser.parseFromString(htmlBruto, "text/html"); // No Letterboxd, os filmes na lista têm a classe 'poster' e o título fica no atributo 'alt' da imagem const posters = doc.querySelectorAll('.poster img'); todosFilmes = []; posters.forEach(img => { const titulo = img.getAttribute('alt'); if (titulo) { todosFilmes.push(titulo); } }); if (todosFilmes.length === 0) { alert("Não conseguimos encontrar filmes. Garante que a tua Watchlist tem filmes visíveis publicamente!"); } else { sortearFilmes(); } } catch (erro) { console.error(erro); alert("Erro ao aceder à página do Letterboxd. Tenta clicar novamente."); } finally { document.getElementById('loading').style.display = 'none'; } }
